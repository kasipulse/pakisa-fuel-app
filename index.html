    async function loadDriversAndVehicles() {
      try {
        // Fetch drivers
        dbg("Fetching drivers...", null);
        const { data: drivers, error: driversError } = await supabase
          .from("drivers")
          .select("*")
          .limit(10);
        dbg("driversError", driversError);
        dbg("drivers", drivers);

        const driverSelect = document.getElementById("driver");
        driverSelect.innerHTML = "";
        if (driversError || !drivers || drivers.length === 0) {
          driverSelect.innerHTML = ``;
        } else {
          drivers.forEach(d => {
            const option = document.createElement("option");
            option.value = d.id;
            option.textContent = `${d.name} (${d.phone})`;
            driverSelect.appendChild(option);
          });
        }

        // Fetch vehicles
        dbg("Fetching vehicles...", null);
        const { data: vehicles, error: vehiclesError } = await supabase
          .from("vehicles")
          .select("*")
          .limit(10);
        dbg("vehiclesError", vehiclesError);
        dbg("vehicles", vehicles);

        const vehicleSelect = document.getElementById("vehicle");
        vehicleSelect.innerHTML = "";
        if (vehiclesError || !vehicles || vehicles.length === 0) {
          vehicleSelect.innerHTML = ``;
        } else {
          vehicles.forEach(v => {
            const option = document.createElement("option");
            option.value = v.id;
            option.textContent = v.registration || v.reg_number;
            vehicleSelect.appendChild(option);
          });
        }

      } catch (err) {
        dbg("loadDriversAndVehicles error", err);
        document.getElementById("status").innerText = "❌ Error loading data - check console";
      }
    }

    document.getElementById("voucherForm").addEventListener("submit", async (e) => {
      e.preventDefault();
      const driver = document.getElementById("driver").value;
      const vehicle = document.getElementById("vehicle").value;
      const amount = parseFloat(document.getElementById("amount").value);

      if (!driver || !vehicle || !amount) {
        document.getElementById("status").innerText = "Please fill all fields.";
        return;
      }

      const { error } = await supabase.from("fuel_vouchers").insert([
        { driver_id: driver, vehicle_id: vehicle, amount_rand: amount }
      ]);

      if (error) {
        document.getElementById("status").innerText = "❌ Error: " + error.message;
        dbg("Insert error", error);
        return;
      }

      document.getElementById("status").innerText = "✅ Voucher created successfully!";
      document.getElementById("voucherForm").reset();
    });

    loadDriversAndVehicles();
